Create Executable jar file
==========================
1.create java class
# touch MyClass.java
-- for class in package "com.example"
# touch com/example/MyClass.java

2.write code into java file
public class MyClass {
    public static void main(String[] args) {
        System.out.println("Hello, world!");
    }
}
-- for class in package "com.example"
package com.example;
public class MyClass {
    public static void main(String[] args) {
        System.out.println("Hello, world!");
    }
}


3.compile java file to generate MyClass.class
# javac MyClass.java

4.create jar with class file
-- for class in package "com.example"
# javac com/example/MyClass.java
# jar cfe myapp.jar com.example.MyClass com/example/
# javac -d appdir com/example/*.java
# jar cfe myapp.jar com.example.MyClass -C appdir .
