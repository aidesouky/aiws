create or replace PROCEDURE INSERT_DATA (
 V_XML_CLOB CLOB
) AS
 V_XML XMLTYPE;
BEGIN
 V_XML := XMLTYPE(V_XML_CLOB);
 FOR REC IN (
  SELECT
   X.ID,
   X.NAME
  FROM
   XMLTABLE ( '/DATA_DS/G_1'
     PASSING V_XML
    COLUMNS
     ID NUMBER PATH 'ORGANIZATION_ID',
     NAME VARCHAR2(100) PATH 'ORGANIZATION_NAME'
   ) X
 ) LOOP
  INSERT INTO T_ORGANIZATIONS (
   ORG_ID,
   ORG_NAME
  ) VALUES (
   REC.ID,
   REC.NAME
  );
  COMMIT;
 END LOOP;
END;
